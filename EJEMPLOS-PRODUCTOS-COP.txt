# ============================================
# NOTAS IMPORTANTES SOBRE PRECIOS COP
# ============================================

1. ✅ Los precios NO llevan decimales (COP no tiene centavos)
2. ✅ Usa números enteros: 50000, 1500000, 25000000
3. ❌ NO uses: 50000.00, 1500000.50
4. ✅ Mínimo precio: 1 (un peso colombiano)
5. ✅ Máximo precio: 999999999999 (casi un billón)
6. 💡 Para mostrar al usuario: usa formato con puntos
   - En BD: 1500000
   - Al mostrar: $1.500.000 COP
7. 💡 Salario mínimo en Colombia (2024): ~$1.300.000

# ============================================
# FORMATO DE VISUALIZACIÓN (FRONTEND)
# ============================================

// En el frontend, puedes formatear así:
const formatCOP = (price) => {
  return new Intl.NumberFormat('es-CO', {
    style: 'currency',
    currency: 'COP',
    minimumFractionDigits: 0,
    maximumFractionDigits: 0
  }).format(price);
};

// Ejemplos:
formatCOP(1500000)  // → "$1.500.000"
formatCOP(50000)    // → "$50.000"
formatCOP(3500000)  // → "$3.500.000"
